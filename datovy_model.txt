// Use DBML to define your database structure
// Docs: https://dbml.dbdiagram.io/docs

Table users {
  id integer [primary key] 
  first_name varchar
  last_name varchar
  nickname varchar
  email varchar
  password varchar
  color varchar
}

Table channel_users {
  id integer [primary key]
  user_id integer
  channel_id integer
  role varchar  // admin or user
}

Table channels {
  id integer [primary key]
  name varchar
  color varchar
  status varchar  // public or private
}
Table messages {
  id integer [primary key]
  user_id integer
  channel_id integer
  msg_text varchar
}

Ref : channel_users.user_id > users.id // many-to-one
Ref : channel_users.channel_id > channels.id // many-to-one
Ref : messages.channel_id > channels.id
